    // DSR Calculation Logic
    $storedRate = (float)($loanModel->plan->interest ?? 54);
    $calcDsr = function($amt, $ten, $rate, $cap) {
        if($amt <= 0 || $ten <= 0 || $cap <= 0) return 0;
        $mRate = ($rate > 20) ? ($rate / 12 / 100) : ($rate / 100);
        if($mRate == 0) $inst = $amt / $ten;
        else $inst = ($amt * $mRate * pow(1+$mRate, $ten)) / (pow(1+$mRate, $ten) - 1);
        return ($inst / $cap) * 100;
    };

    // Repayment Capacity
    $hud_sales = $engine_results['forensics']['considered_sales'] ?? 0;
    $hud_cogs = $derivedCogs ?? 0;
    $hud_grossProfit = $hud_sales - $hud_cogs;
    
    $bizExp = $total_business_expenses ?? $this->total_business_expenses ?? 0;
    $famExp = $total_family_expenses ?? $this->total_family_expenses ?? 0;
    $bufRate = $unforeseen_buffer_rate ?? $this->unforeseen_buffer_rate ?? 0.20;
    $otherLoans = $other_bank_installments ?? $this->other_bank_installments ?? 0;
    
    $totalBuffered = (($bizExp + $famExp) * (1 + $bufRate)) + $otherLoans;
    $hud_repaymentCapacity = max(0, $hud_grossProfit - $totalBuffered);

    // 1. Customer Request 
    $req_amt = $loanModel->amount ?? 0;
    $req_ten = $loanModel->duration ?? 0;
    $req_dsr = $calcDsr($req_amt, $req_ten, $storedRate, $hud_repaymentCapacity);
    
    // 2. LO Suggestion
    $lo_amt = (float)str_replace(',', '', $lo_suggested_amount ?? $suggested_amount ?? 0);
    $lo_ten = (int)($lo_suggested_tenor ?? $suggested_tenor ?? 0);
    $lo_dsr = $calcDsr($lo_amt, $lo_ten, $storedRate, $hud_repaymentCapacity);

    // 3. BM Recommendation
    $bm_amt = (float)($bm_approved_amount ?? $bm_recommended_amount ?? 0);
    $bm_ten = (int)($bm_approved_tenor ?? $bm_recommended_tenor ?? 0);
    $bm_dsr = $calcDsr($bm_amt, $bm_ten, $storedRate, $hud_repaymentCapacity);
    
    // 4. Current Final Structure
    $hoc_amt = (float)str_replace(',', '', $hoc_structure_amount ?? 0);
    $hoc_ten = (int)($hoc_structure_tenor ?? 0);
    $hoc_rate = (float)($hoc_interest_rate ?? $storedRate);

        if($mRate == 0) $hoc_pmt = $hoc_amt / $hoc_ten;
        else $hoc_pmt = ($hoc_amt * $mRate * pow(1+$mRate, $hoc_ten)) / (pow(1+$mRate, $hoc_ten) - 1);
    }
